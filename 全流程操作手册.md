**操作步骤 1.2 - 生成模拟数据 (带异常)**

创建一个名为 `generate_data.py`的文件，写入以下代码来生成包含“脏数据”的原始数据集：

```
import pandas as pd
import numpy as np

np.random.seed(42)  # 固定随机种子，让每次生成的数据一样

n_students = 100
data = {
    'student_id': range(1001, 1001 + n_students),
    'name': [f'学生{i}' for i in range(1, n_students + 1)],
    'gender': np.random.choice(['Male', 'Female', 'M', 'F', '男', '女'], n_students),  # 格式不一致
    'chinese': np.round(np.random.normal(85, 20, n_students), 1),  # 均分85，标准差20
    'math': np.round(np.random.normal(80, 25, n_students), 1),
    'english': np.round(np.random.normal(90, 15, n_students), 1),
    'physics': np.round(np.random.normal(70, 18, n_students), 1),
}
df = pd.DataFrame(data)

# === 主动插入“脏数据” ===
# 1. 插入缺失值
df.loc[np.random.choice(df.index, 5), 'math'] = np.nan
df.loc[np.random.choice(df.index, 3), 'gender'] = np.nan

# 2. 插入异常值 (错误的分数)
df.loc[0, 'chinese'] = 200   # 分数超过150
df.loc[1, 'physics'] = -5    # 负分数
df.loc[2, 'english'] = '缺考' # 非数字类型

# 3. 插入重复行
df = pd.concat([df, df.iloc[[3, 4]]], ignore_index=True)

# 保存
df.to_csv('scores.csv', index=False, encoding='utf-8')
print("✅ 模拟数据（含脏数据）已生成到 scores.csv")
```

- 

  **思路**：先生成100条正常数据，然后主动“污染”它，制造缺失值(`np.nan`)、异常值(200, -5)、格式错误(`‘缺考’`)、重复行，用于后续清洗练习。运行此文件即可得到你的原始数据`scores.csv`。

import pandas as pd

def load_and_inspect(file_path):
    """加载数据并初步查看"""
    df = pd.read_csv(file_path, encoding='utf-8')
    print("1. 数据形状 (行数, 列数):", df.shape)
    print("\n2. 前5行数据:")
    print(df.head())
    print("\n3. 数据基本信息:")
    print(df.info())
    print("\n4. 检查缺失值:")
    print(df.isnull().sum())
    return df

if __name__ == '__main__':
    raw_df = load_and_inspect('scores.csv')
```

- 

  **代码解释**：`pd.read_csv`读取文件。`df.head()`查看头部。`df.info()`查看列类型和内存。`df.isnull().sum()`统计每列空值数量。这是清洗前的“体检报告”。

**操作步骤 2.2 - 系统化数据清洗思路与实现**

创建 `data_cleaner.py`，并依次实现以下函数，形成清洗流水线：

1. 

   **处理缺失值**：

```
def handle_missing_values(df):
    """填充或删除缺失值"""
    # 成绩列用中位数填充（中位数比平均值更抗异常值干扰）
    score_cols = ['chinese', 'math', 'english', 'physics']
    for col in score_cols:
        median_val = df[col].median()  # 先计算中位数
        df[col].fillna(median_val, inplace=True) # fillna填充
    # 性别列用众数（最常出现的值）填充
    if 'gender' in df.columns:
        df['gender'].fillna(df['gender'].mode()[0], inplace=True)
    print("缺失值处理完成。")
    return df
```

1. 

   **处理异常值与格式错误**：

```
def handle_abnormal_values(df):
    """修正超出范围的成绩和格式"""
    # 定义合理的成绩范围
    score_ranges = {'chinese': (0, 150), 'math': (0, 150), 'english': (0, 150), 'physics': (0, 100)}
    for col, (low, high) in score_ranges.items():
        # 将非数字或超出范围的值设为NaN
        df[col] = pd.to_numeric(df[col], errors='coerce')  # '缺考'会变成NaN
        out_of_range = (df[col] < low) | (df[col] > high)
        df.loc[out_of_range, col] = np.nan
        # 再次用中位数填充这些NaN
        df[col].fillna(df[col].median(), inplace=True)
    # 性别列标准化
    gender_map = {'M': 'Male', 'F': 'Female', '男': 'Male', '女': 'Female'}
    df['gender'] = df['gender'].map(gender_map).fillna('Male')  # 未映射的默认处理
    print("异常值与格式标准化完成。")
    return df
```

1. 

   **处理重复数据**：

```
def handle_duplicates(df):
    """删除完全相同的行"""
    before = len(df)
    df.drop_duplicates(inplace=True)  # 直接删除重复行
    after = len(df)
    print(f"删除了 {before - after} 条重复记录。")
    return df
```

1. 

   **计算衍生列**：

```
def add_calculated_columns(df):
    """计算总分、平均分"""
    df['total_score'] = df[['chinese', 'math', 'english']].sum(axis=1)  # axis=1按行加
    df['avg_score'] = df['total_score'] / 3
    print("已计算总分和平均分。")
    return df
```

1. 

   **主清洗函数**：

```
def clean_data_pipeline(raw_df):
    """执行完整的清洗流程"""
    print("开始数据清洗流程...")
    df = raw_df.copy()
    df = handle_missing_values(df)
    df = handle_abnormal_values(df)
    df = handle_duplicates(df)
    df = add_calculated_columns(df)
    # 保存清洗后的数据
    df.to_csv('clean_scores.csv', index=False, encoding='utf-8')
    print("✅ 清洗完成，干净数据已保存为 clean_scores.csv")
    return df
```

import pandas as pd
clean_df = pd.read_csv('clean_scores.csv')

# 1. 整体描述（各科的平均分、标准差、最高分等）

def plot_subject_avg(clean_df):
    subjects = ['语文', '数学', '英语', '物理']
    averages = [clean_df['chinese'].mean(), clean_df['math'].mean(), clean_df['english'].mean(), clean_df['physics'].mean()]
    plt.figure(figsize=(10,6))
    bars = plt.bar(subjects, averages, color=['skyblue', 'lightgreen', 'orange', 'pink'])
    plt.title('各科平均分对比')
    plt.ylabel('平均分')
    # 在柱子上方添加数值
    for bar in bars:
        height = bar.get_height()
        plt.text(bar.get_x() + bar.get_width()/2., height, f'{height:.1f}', ha='center', va='bottom')
    plt.savefig('各科平均分.png')  # 保存图片
    plt.show()
```

def plot_correlation_heatmap(clean_df):
    plt.figure(figsize=(8,6))
    # 计算相关系数矩阵
    corr = clean_df[['chinese', 'math', 'english', 'physics']].corr()
